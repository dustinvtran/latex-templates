\begin{algorithm}[tb]
  \caption{Bubble Sort}
  \label{alg:example}
\begin{algorithmic}
  \STATE {\bfseries Input:} data $x_i$, size $m$
  \REPEAT
  \STATE Initialize $noChange = true$.
  \FOR{$i=1$ {\bfseries to} $m-1$}
    \IF{$x_i > x_{i+1}$}
      \STATE Swap $x_i$ and $x_{i+1}$
      \STATE $noChange = false$
    \ENDIF
  \ENDFOR
  \UNTIL{$noChange$ is $true$}
\end{algorithmic}
\end{algorithm}
